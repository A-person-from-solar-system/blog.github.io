package com.coda.result;

import java.io.Serializable;

public class Result<T> implements Serializable {
    private Integer code; //编码：1成功，0和其它数字为失败
    private String msg; //错误信息
    private T data; //数据

    public static <T> Result<T> success() {  // 返回成功数据(不带参数)
        Result<T> result = new Result<T>();
        result.code = 1;
        return result;
    }

    public static <T> Result<T> success(T object) {  // 返回成功数据(带参数)
        Result<T> result = new Result<T>();
        result.data = object;
        result.code = 1;
        return result;
    }

    public static <T> Result<T> error(String msg) {  // 返回失败数据
        Result result = new Result();
        result.msg = msg;
        result.code = 0;
        return result;
    }
}
